	public void TesteMD5() throws NoSuchAlgorithmException {
		String senha="admin";
		MessageDigest m = MessageDigest.getInstance("MD5");
		m.update(senha.getBytes(),0,senha.length());
		System.out.println("MD5: " + new BigInteger(1,m.digest()).toString(16));
